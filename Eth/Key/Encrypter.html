def data
  {
    crypto: {
      cipher: cipher_name,
      cipherparams: {
        iv: bin_to_hex(iv),
      },
      ciphertext: bin_to_hex(encrypted_key),
      kdf: "pbkdf2",
      kdfparams: {
        c: iterations,
        dklen: 32,
        prf: prf,
        salt: bin_to_hex(salt),
      },
      mac: bin_to_hex(mac),
    },
    id: id,
    version: 3,
  }.tap do |data|
    data[:address] = address unless options[:skip_address]
  end
end
